# Cloudflare Pages Deployment Configuration
# Скопируйте этот файл в .env и заполните значения

# Cloudflare Account ID
# Найти: https://dash.cloudflare.com → справа внизу "Account ID"
CF_ACCOUNT_ID=your_account_id_here

# Cloudflare API Token
# Создать: https://dash.cloudflare.com/profile/api-tokens
# Нужны права: Account → Cloudflare Pages → Edit
CF_API_TOKEN=your_api_token_here

# Имя проекта Cloudflare Pages (опционально, по умолчанию: dashboardtt)
CF_PROJECT_NAME=dashboardtt
